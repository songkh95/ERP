                const qty = parseFloat(row.querySelector('.qty-input').value) || 0;
                const price = parseFloat(row.querySelector('.price-input').value) || 0;
                const supplyPrice = qty * price;
                row.querySelector('.supply-price').value = this.formatNumber(supplyPrice);
                this.calculateTotal();
            },

            calculateTotal: function() {
                const tbody = document.getElementById('itemTableBody');
                let totalSupply = 0;
                Array.from(tbody.rows).forEach(row => {
                    const val = row.querySelector('.supply-price').value.replace(/,/g, '');
                    totalSupply += parseFloat(val) || 0;
                });
                const vat = Math.floor(totalSupply * 0.1);
                const grand = totalSupply + vat;

                document.getElementById('subTotal').value = this.formatNumber(totalSupply);
                document.getElementById('vatTotal').value = this.formatNumber(vat);
                document.getElementById('grandTotal').value = this.formatNumber(grand);
                const koreanTotalElement = document.getElementById('koreanTotal');
                if(koreanTotalElement) koreanTotalElement.innerText = this.numberToKorean(grand);
            },

            formatNumber: function(num) {
                return num.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
            },

            numberToKorean: function(number) {
                if (number === 0) return "ì˜";
                return this.formatNumber(number); 
            }
        };

        Estimate.init();
        window.Estimate = Estimate;
    </script>
